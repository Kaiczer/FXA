    id = annex.1287
    title = annexation_ALG_CONQ
    desc = annexation_ALG_CONQ_desc
    picture = GFX_report_event_generic_conference2
    is_triggered_only = yes
    immediate = {
        hidden_effect = {
            add_to_temp_array = { core_removal_array = ALG }
            remove_all_cores_and_claims_of_tags = yes
            every_state = {
                limit = {
                    state = 805
                }
                add_core_of = ALG
            }
        }
    }
    option = { # Liberate ALG
        log = "[GetLogInfo]: event annex.1287 option annex.1287.a"
        name = annex.1287.a
        ai_chance = { base = 100 }
        set_temp_variable = { tag_to_release = ALG }
        release_targeted_tag = yes
        recheck_annexations = yes
    }
    option = { # Occupy ALG
        log = "[GetLogInfo]: event annex.1287 option annex.occupation"
        name = annex.occupation
        ai_chance = { base = 0 }
        occupy_replace_country = yes
        set_country_flag = ALG_CONQ
        recheck_annexations = yes
    }
}
