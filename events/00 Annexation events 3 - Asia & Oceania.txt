        name = annex.1147.a

        set_temp_variable = { tag_to_release = BRM }
        release_targeted_tag = yes
        recheck_annexations = yes
    }
    option = { # Occupy BRM
        log = "[GetLogInfo]: event annex.1147 option annex.occupation"
        name = annex.occupation
        ai_chance = { base = 0 }
        occupy_replace_country = yes
        set_country_flag = BRM_CONQ
        recheck_annexations = yes
    }
}
