# Pastaza - annexation_PASTAZA_CONQ
can_release_pastaza = {
    state = 1051
    NOT = { is_claimed_by = ROOT }
    ROOT = {
        should_release_ecuador_first = yes
        should_release_peru_first = yes 
        OR = {
            AND = {
                original_tag = PRU
                owns_state = 1052
            }
            original_tag = ECU
            ECU = { ROOT_can_grant_land = yes }
            var:1052.owner = {
                original_tag = PRU
                ROOT_can_grant_land = yes 
            }
        }
        NOT = {
            has_country_flag = PASTAZA_CONQ
        }
    }
}

# Arica - annexation_ARICA_CONQ
can_release_arica = {
    state = 1065
    NOT = { is_claimed_by = ROOT }
    ROOT = {
        should_release_bolivia_first = yes
        should_release_peru_first = yes
        should_release_chile_first = yes 
        if = {
            limit = { 
                OR = {
                    tag = BOL
                    BOL = { is_ally_of_ROOT = yes }
                }
            }
            OR = {
                1066 = { is_valid_annexation_state = no }
                has_country_flag = TARAPACA_CONQ
            }
        }
        OR = {
            AND = {
                OR = {
                    original_tag = BOL
                    original_tag = CHL
                }
                owns_state = 1066
            }
            original_tag = PRU
            ROOT = { has_cosmetic_tag = PBC }
            var:1066.owner = {
                OR = {
                    has_cosmetic_tag = PBC
                    original_tag = BOL
                    original_tag = CHL
                }
                ROOT_can_grant_land = yes 
            }
            PRU = { ROOT_can_grant_land = yes }
        }
        NOT = {
            has_country_flag = ARICA_CONQ
        }
    }
}
